/* 1. 같은 좌표의 모든 대사를 찾아 순차 실행하는 로직 */ async function updateState() { const posStr = `${playerPos.x}-${playerPos.y}`; // [변경] find 대신 filter를 사용하여 해당 좌표의 모든 행을 가져옵니다. const events = scenarioData.filter(row => row.Coord === posStr); if (events.length > 0 && !isTyping) { // 첫 번째 행에서 구역 정보를 가져와 헤더 업데이트 const firstEv = events[0]; const area = scenarioData.find(row => row.AreaID === firstEv.AreaID && !row.Coord); if (area) { document.getElementById('area-name').innerText = area.AreaName; document.getElementById('area-desc').innerText = area.AreaDesc; } // 모든 대사 행을 순차적으로 타자기 출력 await playDialogueSequence(events); document.getElementById('dist-val').innerText = String(totalDistance).padStart(3, '0'); } } async function playDialogueSequence(events) { isTyping = true; setButtonsDisabled(true); for (const ev of events) { const text = ev.Content.trim(); if (text) { await renderSingleBox(text); // 대사 사이의 짧은 호흡 (0.4초) await new Promise(res => setTimeout(res, 400)); } } isTyping = false; // 마지막 행의 데이터를 기준으로 버튼 활성화 updateButtons(events[events.length - 1]); } async function renderSingleBox(text) { const logContent = document.getElementById('log-content'); const box = document.createElement('div'); // 태그별 박스 스타일 결정 if (text.startsWith("[SYSTEM]")) box.className = 'msg-box sys-box'; else if (text.startsWith("[NOISE]") || text.startsWith("[DATA]")) box.className = 'msg-box noise-box'; else box.className = 'msg-box analyst-box'; logContent.appendChild(box); const textSpan = document.createElement('span'); box.appendChild(textSpan); const cursor = document.createElement('span'); cursor.className = 'cursor'; box.appendChild(cursor); for (let char of text) { textSpan.innerText += char; document.getElementById('log-container').scrollTop = document.getElementById('log-container').scrollHeight; await new Promise(res => setTimeout(res, 25)); } cursor.remove(); }
